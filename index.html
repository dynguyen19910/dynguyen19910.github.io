Sample SQL 
-- Top 10 customers by profit (last 12 months)
WITH sales AS (
  SELECT order_id, customer_id, order_date, profit
  FROM orders
  WHERE order_date >= DATE('now','-12 months')
)
SELECT c.customer_name, SUM(s.profit) AS total_profit
FROM sales s
JOIN customers c ON c.customer_id = s.customer_id
GROUP BY 1
ORDER BY total_profit DESC
LIMIT 10;

import pandas as pd
flights = pd.read_csv('flights.csv')
flights['flight_date'] = pd.to_datetime(flights['flight_date'])

by_carrier = (flights
  .query("flight_date.dt.month.isin([12,1,2])", engine='python')
  .groupby('carrier')['arr_delay']
  .mean()
  .sort_values())

Sample Python (cleaner) 
import pandas as pd
from pathlib import Path

def clean_csv(path_in, path_out):
    df = pd.read_csv(path_in)
    for col in df.columns:
        if 'date' in col.lower():
            df[col] = pd.to_datetime(df[col], errors='coerce')
    for col in df.select_dtypes('object').columns:
        df[col] = df[col].astype(str).str.strip().str.title()
    df = df.drop_duplicates()
    df.to_csv(path_out, index=False)

for p in Path('incoming').glob('*.csv'):
    clean_csv(p, Path('clean')/p.name)
