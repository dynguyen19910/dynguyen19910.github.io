import pandas as pd
def clean(df):
  for c in df.columns:
    if 'date' in c.lower():
      df[c] = pd.to_datetime(df[c], errors='coerce')
  for c in df.select_dtypes('object'):
    df[c] = df[c].astype(str).str.strip().str.title()
  return df.drop_duplicates()
